int main() {

  cin >> n >> t;
  long long a[n];
  for(int i = 0; i < n; i++){
    cin >> a[i];
  }
  vector<long long> s(n+1,0);
  for(int i = 1; i <= n; i++){
    s[i] = s[i-1] + a[i-1];
  }
  long long sum = 0;
  for(int i = 0; i < n; i++){
    sum += a[i];
  }
  long long now = t % sum;

  for(int i = 1; i <= n; i++){
    if(s[i-1] <= now && now <= s[i]){
      cout << i << " " << now - s[i-1] << endl;
      return 0;
    }
  }

}
