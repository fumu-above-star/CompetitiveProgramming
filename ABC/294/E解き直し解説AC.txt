int combination(int n, int r) {
  if ( r * 2 > n ) r = n - r;
  int dividend = 1;
  int divisor  = 1;
  for (  int i = 1; i <= r; ++i ) {
    dividend *= (n-i+1);
    divisor  *= i;
  }
  return dividend / divisor;
}
