int main(){//s‚ðt‚É
    int H,W;

    vector<string> s;
    vector<string> t;
  	s.resize(H);
  	t.resize(H);
    for(int h = 0; h < H; h++){
        cin >> s[h];
    }
    for(int h = 0; h < H; h++){
        cin >> t[h];
    }
    
    vector<int> scnt(H,0);
    vector<int> tcnt(H,0);
    
  	for(int h = 0; h < H; h++){
        for(int w = 0; w < W; w++){
            if(s[h][w] == '#'){
                scnt[h]++;
            }
        }
    }
 
    for(int h = 0; h < H; h++){
        for(int w = 0; w < W; w++){
            if(t[h][w] == '#'){
                tcnt[h]++;
            }
        }
    }
 
    int flag = 1;
    for(int h = 0; h < H; h++){
        if(tcnt[h] != scnt[h]){
            flag = 0;
            break;
        }
    }
 
    if(flag){
        cout << "Yes" << endl;
    }else{
        cout << "No" << endl;
    }
}
